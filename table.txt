Query
SELECT
    req.first_name, 
    req.last_name, 
    req.recipient_email, 
    req.requester_email, 
    req.phone_number, 
    dev.device_name, 
    dev.device_type, 
    dev.device_id, 
    inv.user_associatedid, 
    inv.device_type, 
    inv.device_model, 
    inv.device_name
FROM requests req
JOIN devices dev
ON CAST(REGEXP_REPLACE(SPLIT_PART(req.device_quantities::TEXT, ':', 1), '[^0-9]', '', 'g') AS INTEGER) = dev.device_id
JOIN inventory inv
ON req.replaced_item_id = inv.item_id;
